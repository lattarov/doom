(defun vlv-format-org-elisp-blocks ()
  "Reindent all emacs-lisp src blocks without triggering buffer hooks."
  (interactive)
  (save-excursion
    (goto-char (point-min))
    (while (re-search-forward "^[ \t]*#\\+begin_src +emacs-lisp[^\n]*\n" nil t)
      (let ((content-beg (point)))
        (when (re-search-forward "^[ \t]*#\\+end_src" nil t)
          (let* ((content-end (match-beginning 0))
                 (content (buffer-substring-no-properties content-beg content-end))
                 (indented (with-temp-buffer
                             (let ((prog-mode-hook nil)
                                   (lisp-data-mode-hook nil)
                                   (emacs-lisp-mode-hook nil)
                                   (after-change-major-mode-hook nil)
                                   (change-major-mode-after-body-hook nil))
                               (emacs-lisp-mode))
                             (insert content)
                             (indent-region (point-min) (point-max))
                             (buffer-string))))
            (unless (string= content indented)
              (let ((end-marker (point-marker)))
                (goto-char content-beg)
                (delete-region content-beg content-end)
                (insert indented)                 (goto-char end-marker)))))))))

(add-hook 'org-mode-hook
          (lambda ()
            (add-hook 'before-save-hook #'vlv-format-org-elisp-blocks nil t)))

#+begin_src emacs-lisp :tangle config.el
(after! (:and eww org)
  (defun jao-eww-to-org (&optional dest)
    "Render the current eww buffer using org markup.
If DEST, a buffer, is provided, insert the markup there."
    (interactive)
    (unless (org-region-active-p)
      (let ((shr-width 80)) (eww-readable)))
    (let* ((start (if (org-region-active-p) (region-beginning) (point-min)))
           (end (if (org-region-active-p) (region-end) (point-max)))
           (buff (or dest (generate-new-buffer "*eww-to-org*")))
           (link (eww-current-url))
           (title (or (plist-get eww-data :title) "")))
      (with-current-buffer buff
        (insert "#+title: " title "\n#+link: " link "\n\n")
        (org-mode))
      (save-excursion
        (goto-char start)
        (while (< (point) end)
          (let* ((p (point))
                 (props (text-properties-at p))
                 (k (seq-find (lambda (x) (plist-get props x))
                              '(shr-url image-url outline-level face)))
                 (prop (and k (list k (plist-get props k))))
                 (next (if prop
                           (next-single-property-change p (car prop) nil end)
                         (next-property-change p nil end)))
                 (txt (buffer-substring (point) next))
                 (txt (replace-regexp-in-string "\\*" "·" txt)))
            (with-current-buffer buff
              (insert
               (pcase prop
                 ((and (or `(shr-url ,url) `(image-url ,url))
                       (guard (string-match-p "^http" url)))
                  (let ((tt (replace-regexp-in-string "\n\\([^$]\\)" " \\1" txt)))
                    (org-link-make-string url tt)))
                 (`(outline-level ,n)
                  (concat (make-string (- (* 2 n) 1) ?*) " " txt "\n"))
                 ('(face italic) (format "/%s/ " (string-trim txt)))
                 ('(face bold) (format "*%s* " (string-trim txt)))
                 (_ txt))))
            (goto-char next))))
      (pop-to-buffer buff)
      (goto-char (point-min)))))

(use-package! yasnippet
  :defer t
  :config
  (defun yasnippet-extract-arg-and-type (arg)
    "Extract argument name and type from a string of the form `arg: type'."
    (let* ((parts (split-string arg ": ")))
      (if (= (length parts) 2)
          (cons (car parts) (cadr parts))
        (cons (car parts) "type"))))

  (defun yasnippet-python-params-doc (arg-string)
    "Generate NumPy-style docstring for parameters with optional type annotations.
If there are no arguments, return an empty string."
    (let* ((args (split-string arg-string ", "))
           (formatted-doc
            (if (string-blank-p arg-string)
                ""
              (mapconcat (lambda (arg)
                           (let* ((arg-pair (yasnippet-extract-arg-and-type arg))
                                  (arg-name (car arg-pair))
                                  (arg-type (cdr arg-pair)))
                             ;; TODO: Generate placeholders for each parameter description.
                             (format "%s : %s\n        TODO: Description of %s." arg-name arg-type arg-name)))
                         args
                         "\n    "))))
      formatted-doc))

  (defun yasnippet-python-parameter-assignments (arg-string)
    "Convert a comma-separated string of arguments into self assignments.
If there are no arguments, return `pass'."
    (let* ((args (split-string arg-string ", "))
           (assignments
            (if (string-blank-p arg-string)
                "pass"
              (mapconcat (lambda (arg)
                           (let* ((parts (split-string arg ": "))
                                  (name (car parts)))
                             (format "self.%s = %s" name name)))
                         args
                         "\n        "))))
      assignments)))

(defvar-local vlv-yas-vterm-had-word nil
  "Whether there was a word at point when snippet expansion started.")

(defun vlv-vterm-yas-confirm ()
  "Send expanded snippet from temp buffer to vterm."
  (interactive)
  (let ((result (buffer-string))
        (vterm-buf vlv-yas-vterm-source-buf)
        (had-word vlv-yas-vterm-had-word)
        (win (selected-window)))
    (kill-buffer (current-buffer))
    (ignore-errors (delete-window win))
    (pop-to-buffer vterm-buf)
    (when had-word (vterm-send-meta-backspace))
    (vterm-send-string result)))

(defun vlv-vterm-yas-expand ()
  "Expand yasnippet in a popup buffer.
 If the snippet has no placeholders, send it to vterm immediately
 without displaying the temp buffer."
  (interactive)
  (let* ((word (thing-at-point 'word 'no-properties))
         (vterm-buf (current-buffer))
         (tmp-buf (generate-new-buffer "*yas-vterm*")))
    (with-current-buffer tmp-buf
      (sh-mode)
      (yas-minor-mode 1)
      (setq vlv-yas-vterm-source-buf vterm-buf)
      (setq vlv-yas-vterm-had-word word)
      (when word (insert word))
      (if (yas-expand)
          ;; yas-expand returned non-nil, a snippet was triggered.
          ;; Now check whether it left any active placeholders.
          (if (yas-active-snippets)
              ;; Has placeholders: show the popup for the user to fill in.
              (progn
                (when (bound-and-true-p evil-mode)
                  (evil-insert 1))
                (define-key (current-local-map) (kbd "Z Z") #'vlv-vterm-yas-confirm)
                (define-key (current-local-map) (kbd "Z Q") #'kill-buffer-and-window)
                (setq header-line-format
                      (format "Press %s to send, %s to abandon."
                              (key-description
                               (where-is-internal #'vlv-vterm-yas-confirm
                                                  (current-local-map) t))
                              (key-description
                               (where-is-internal #'kill-buffer-and-window
                                                  (current-local-map) t))))
                (pop-to-buffer tmp-buf
                               '(display-buffer-below-selected
                                 (window-height . 5))))
            ;; No placeholders: grab result and send immediately.
            (let ((result (buffer-string)))
              (kill-buffer tmp-buf)
              (with-current-buffer vterm-buf
                (when word (vterm-send-meta-backspace))
                (vterm-send-string result))))
        ;; yas-expand found nothing: clean up and report.
        (kill-buffer tmp-buf)
        (message "No snippet found for: %s" (or word "<no word at point>"))))))
(map!
 :after vterm
 ;; approach vterm behaviour to eshell
 :map vterm-mode-map
 ;; completion
 :i "<tab>" nil
 :i "<tab>" #'vlv-vterm-yas-expand

 ;; Fix 0   go to start of input, not buffer column 0
 :n "0"  (cmd! (vterm-send-key "a" nil nil t)
               (evil-refresh-cursor))

 ;; Fix $   go to end of input

               (evil-refresh-cursor))

 ;; Fix dd   clear input instead of deleting the buffer line
 ;; unbind "d" so that it can be used as a key sequence.
 :n "d"  nil

 :n "dd" (cmd! (vterm-send-key "e" nil nil t)
               (vterm-send-key "u" nil nil t)
               (evil-refresh-cursor))

 ;; Fix V    select the line from the tty "$" character onwards
 :n "V"  (cmd! (let* ((beg (save-excursion
                             (beginning-of-line)
                             (re-search-forward "\\$ " (line-end-position) t)
                             (point)))
                      (end (save-excursion (vterm-end-of-line) (point))))
                 (evil-visual-make-region beg end))))

(use-package! eww
  :defer t
  :init

  (defconst vlv-eww-workspace-name "*EWW*"
    "Name of the dedicated EWW workspace.")

  (defun vlv-eww--get-or-create-workspace ()
    "Switch to EWW workspace, creating it if needed."
    (if (+workspace-exists-p vlv-eww-workspace-name)
        (+workspace-switch vlv-eww-workspace-name)
      (+workspace-new vlv-eww-workspace-name)
      (+workspace-switch vlv-eww-workspace-name)))

  (defun vlv-eww--workspace-eww-buffers ()
    "Return EWW buffers belonging to the EWW workspace, excluding current buffer."
    (when (+workspace-exists-p vlv-eww-workspace-name)
      (seq-filter
       (lambda (buf)
                                        ; exclude the buffer being killed
         (and (not (eq buf (current-buffer)))
              (eq (buffer-local-value 'major-mode buf) 'eww-mode)))
       (+workspace-buffer-list
        (+workspace-get vlv-eww-workspace-name)))))

  (defun vlv-eww--maybe-close-workspace ()
    "Close EWW workspace if no EWW buffers remain in it."
    (when (+workspace-exists-p vlv-eww-workspace-name)
      (unless (vlv-eww--workspace-eww-buffers)
        (when (string= (+workspace-current-name) vlv-eww-workspace-name)
          (+workspace-switch +workspace--last))
        (persp-kill vlv-eww-workspace-name)
        (message "EWW workspace closed."))))

  (defun vlv-eww (url)
    "Open URL in EWW in the dedicated EWW workspace."
    (interactive
     (list (read-string "URL: " (thing-at-point-url-at-point))))
    (vlv-eww--get-or-create-workspace)
    (eww url)
    (delete-other-windows))

  (defun vlv-eww-open-in-eww-workspace (url &rest _)
    "Open URL in the EWW workspace. Used as `browse-url-browser-function'."
    (vlv-eww url))

    "Close EWW workspace when the last EWW buffer is killed."

#+begin_src emacs-lisp :tangle config.el
(use-package! magit
  :defer t
  :config
  (defun vlv-delete-merged-branches ()
    "Fetch prune then deletes branches already merged to a target branch.
Found in https://emacs.stackexchange.com/questions/60200/magit-remove-local-branches-that-were-merged-into-another-branch"
    (interactive)
    (magit-fetch-all-prune)
    (let* ((default-branch
            (read-string "Default branch: " (magit-get-current-branch)))
           (merged-branches
            (magit-git-lines "branch"
                             "--format" "%(refname:short)"
                             "--merged"
                             default-branch))
           (branches-to-delete
            (remove default-branch merged-branches)))
      (if branches-to-delete
          (if (yes-or-no-p (concat "Delete branches? ["
                                   (mapconcat 'identity branches-to-delete ", ") "]"))
              (apply #'magit-run-git "branch" "-d" branches-to-delete))
        (message "Nothing to delete"))))

  (transient-append-suffix 'magit-branch "C"
    '("K" "delete all merged" vlv-delete-merged-branches))

  (defconst vlv--magit-diff-cache-limit 2000
    "Maximum number of symbols stored per diff buffer in `vlv--magit-diff-cache'.")

  ;; Global accumulated symbol set. Reset on doom/reload or restart.
  (setq vlv--magit-diff-symbols-cache (make-hash-table :test 'equal))

  ;; Tracks the last-seen tick per diff buffer to avoid redundant rescans.
  ;; Weak keys ensure dead buffers are GC'd automatically.
  (setq vlv--magit-diff-cache-buffer-to-tick-tracker (make-hash-table :test 'eq :weakness 'key))

  (defun vlv--split-symbol (sym)
    "Split SYM into parts on hyphens, underscores, and camelCase boundaries."
    (let* ((case-fold-search nil) ; camelCase regex requires case-sensitive matching.
           (parts (split-string
                   (replace-regexp-in-string
                    "\\([a-z]\\)\\([A-Z]\\)" "\\1-\\2" sym)
                   "[-_]+" t)))
      (mapcar #'downcase parts)))

  (defun vlv-magit-diff-capf ()
    "Provide capf completion from symbols seen in any magit-diff buffer.

   Symbols are accumulated globally across all projects for the session,
   and reset only on doom/reload or Emacs restart.
   Symbols containing hyphens, underscores, or camelCase boundaries are
   also split into their constituent parts."
    (when-let ((bounds (bounds-of-thing-at-point 'symbol))
               (buf (magit-get-mode-buffer 'magit-diff-mode)))
      (with-current-buffer buf
        (let ((tick (buffer-modified-tick)))
          (unless (equal tick (gethash buf vlv--magit-diff-cache-buffer-to-tick-tracker))
            ;; add new symbols to cache and update the scan tracker.
            (puthash buf tick vlv--magit-diff-cache-buffer-to-tick-tracker)
            (save-excursion
              (goto-char (point-min))
              (while (re-search-forward "\\_<\\(?:\\sw\\|\\s_\\)\\{3,\\}\\_>" nil t)
                (let ((sym (match-string-no-properties 0)))
                  ;; add the full symbol.
                  (puthash sym t vlv--magit-diff-symbols-cache)
                  ;; add each split part.
                  (dolist (part (vlv--split-symbol sym))
                    (puthash part t vlv--magit-diff-symbols-cache))))))))
      (list (car bounds) (cdr bounds)
            (hash-table-keys vlv--magit-diff-symbols-cache)
            :exclusive 'no)))


  (add-hook! 'git-commit-mode-hook
    (add-hook 'completion-at-point-functions #'vlv-magit-diff-capf nil t))

  (setopt magit-clone-always-transient t)

  (setopt magit-clone-default-directory (if (vlv-env-personal-p)
                                            "~/dev/"
                                          "D:/@Src/"))

  ;; add "develop" to magit long lived and main branch name lists.
  (add-to-list 'magit-long-lived-branches "develop")
  (add-to-list 'magit-main-branch-names "develop")

  ;; work commit format is long by nature...
  (setopt magit-show-long-lines-warning (unless (vlv-env-work-p) t)))
